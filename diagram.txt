  id bigint(20) [PK]
  name varchar(255)
  password varchar(255)
  profile varchar(300)
  bio varchar(256)
  monney int
  remember_token varchar(100) 
  created_at timestamp
  update_at timestamp
}

  update_at timestamp
}
